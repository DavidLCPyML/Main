#include<iostream>
#include<vector>
#include<string>
using namespace std;

long long res(long long k, long long n) {
	if (n % 2 == 0) {
		long long mid = n / 2;
		if (k > mid) {
			return 2 * (k - mid);
		} else {
			return 2 * k - 1;
		}
	} else {
		long long mid = (n + 1) / 2;
		if (k > mid) {
			return 2 * (k - mid);
		} else {
			return 2 * k - 1;
		}
	}
}

int main() {
	long long n, k;
	cin >> n >> k;

	cout << res(k, n);
	return 0;
}
