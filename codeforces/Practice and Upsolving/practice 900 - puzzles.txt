#include<iostream>
#include <vector>
#include <algorithm>
using namespace std;

void solve(int n, int m) {
    vector<int> puzzles;
    for (int i = 0; i < m; i++) {
        int puzzle;
        cin >> puzzle;
        puzzles.push_back(puzzle);
    }
    sort(puzzles.begin(), puzzles.end());

    int mindiff = 999;
    for (int i = 0; i < m; i++) {
        if (i + n <= m) {
            mindiff = min(mindiff, puzzles[i+n-1]-puzzles[i]);
        } else
            break;
    }
    cout << mindiff << endl;
}

int main() {
    int n,m;
    cin >> n >> m;
    
    solve(n, m);
    return 0;
}
