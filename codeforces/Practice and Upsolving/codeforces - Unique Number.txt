#include <iostream>
#include <vector>
#include <algorithm>
#include <tuple>
using namespace std;

int main() {
    int tests;
    cin >> tests;
    while (tests--) {
        int x;
        cin >> x;
        vector<int> ans;
        int sum = 0, last = 9;
        while (sum < x && last > 0) {
            ans.push_back(min(x - sum, last));
            sum += last;
            last--;
        }
        if (sum < x) {
            cout << -1 << "\n";
        }
        else {
            reverse(ans.begin(), ans.end());
            for (int i : ans) {
                cout << i;
            }
            cout << "\n";
        }
    }
    return 0;
}
