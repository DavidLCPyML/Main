#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
using namespace std;

vector<int> Kevin;

int main() {
	int N;
	cin >> N;

	for (int i = 0; i < N; i++) {
		int num;
		cin >> num;
		Kevin.push_back(num);
	}

	sort(Kevin.begin(), Kevin.end());
	
	if (N % 2 == 1) {
		int george = Kevin[(N - 1) / 2];
		for (int i = 0; i < N; i++) {
			Kevin[i] = abs(Kevin[i] - george);
		}
	} else {
		int george = Kevin[N / 2];
		for (int i = 0; i < N; i++) {
			Kevin[i] = abs(Kevin[i] - george);
		}
	}
	/*	for (int i = 0; i < N; i++) {
		cout << Kevin[i] << " ";
	} cout << endl;
	*/

	long long int sum = 0;
	for (int i = 0; i < N; i++) {
		sum += Kevin[i];
	}

	cout << sum << endl;
	return 0;
}
