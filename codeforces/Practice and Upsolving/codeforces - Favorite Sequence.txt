#include <iostream>
#include <vector>
#include <algorithm>
#include <tuple>
using namespace std;

int main() {
    int tests;
    cin >> tests;
    while (tests--) {
        int n;
        cin >> n;
        vector<int> v(n);
        for (int& e : v) {
            cin >> e;
        }
        int left = 0, right = n - 1;
        vector<int> ans(n);
        for (int i = 0; i < n; i++) {
            if (i % 2 == 0) {
                ans[i] = v[left++];
            }
            else {
                ans[i] = v[right--];
            }
        }
        for (int i : ans) {
            cout << i << " ";
        }
        cout << "\n";

    }
    return 0;
}
