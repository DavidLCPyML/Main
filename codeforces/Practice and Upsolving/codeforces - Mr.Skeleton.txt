#include <iostream>
#include <vector>
#include <algorithm>
#include <string>
using namespace std;

int main() {
	int N, H;
	cin >> N >> H;

	vector<bool> bones(N+1, false);
	for (int i = 0; i < H; i++) {
		int index;
		cin >> index;

		bones[index] = true;
	}

	for (int i = 1; i < N+1; i++) {
		if (bones[i] == false) {
			cout << i << endl;
		}
	}

	return 0;
}
