#include <iostream>
#include <vector>
#include <algorithm>
#include <string>
using namespace std;

int main() {
	int t;
	cin >> t;

	while (t--) {
        int n;
        cin >> n;
        int ones = 0, twos = 0;
        
        for (int i = 0; i < n; i++) {
            int weight;
            cin >> weight;
            
            if (weight == 1) {
                ones++;
            } else {
                twos++;
            }
        }
        
        if ((ones + 2 * twos) % 2 != 0) {
            cout << "no" << endl;
        } else {
            int sum = (ones + 2 * twos) / 2;
            
            if ((sum % 2 == 0) || (sum % 2 == 1 && ones != 0)) {
                cout << "yes" << endl;
            } else {
                cout << "no" << endl;
            }
        }
	}
	return 0;
}
