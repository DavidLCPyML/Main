#include <iostream>
#include <vector>
#include <algorithm>
#include <string>
using namespace std;

int main() {
    string s, t;
    long long n, l = 0, ans = 0;
    cin >> n >> s >> t;

    for (int i = 0; i < n - 1; i++) {
        if (s[i] == t[0] && s[i + 1] == t[1]) {
            long long to = n - (i + 1); 
            long long from = i - l + 1;
            
            ans += from * to;
            l = i + 1;
        }
    }
    cout << ans << endl;

	return 0;
}
