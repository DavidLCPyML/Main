#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

void solve() {
    int n;
    vector<int> nums;
    cin >> n;
    for (int i = 0; i < n; i++) {
        int x;
        cin >> x;
        nums.push_back(x);
    }
    
    vector<int> ans;
    for (int i = 0; i < nums.size(); i++) {
        ans.push_back(nums[i]);
    }

    int i;
    sort(nums.begin(), nums.end());

    for (i = 0; i < n; i++) {
        auto it = find(ans.begin(), ans.end(), nums[i]);
        //cout << *it << endl;
        int index = it - ans.begin();
        //cout << ans[index] << endl;

        //whether we should perform a switch
        if (abs(index - i) % 2 == 0) {
            ans[index] = -100;
        }
        else {
            break;
        }
    }

    if (i == n)
        cout << "YES" << endl;
    else
        cout << "NO" << endl;
}


int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(NULL);

    int t;
    cin >> t;
    while (t--) {
        solve();
    }
    return 0;
}
