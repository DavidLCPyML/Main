#include <iostream>
#include <vector>
#include <algorithm>
#include <string>
using namespace std;

int main() {
	long long int A, S;
	cin >> A >> S;

    S -= A;
    if (S <= 0) {
        cout << "NO";
        return 0;
    }
    string s = to_string(A);
    string k = to_string(S);

    sort(s.begin(), s.end());
    sort(k.begin(), k.end());

    if (k == s) {
        cout << "YES" << endl;
    } 
    else {
        cout << "NO" << endl;
    }

	return 0;
}
