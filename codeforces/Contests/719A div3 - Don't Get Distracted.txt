#include <iostream>
using namespace std;

bool distracted(string s) {
    int check[26] = { 0 };
    char prev = 'z';

    for (int i = 0; i < s.length();i++) {
        int n = s[i] - 'A';

        if (check[n] > 0) {
            if (prev == s[i])
                check[n]++;
            else
                return false;
        } else
            check[n]++;
        prev = s[i];
    }

    return true;
}

int main() {
    int t;
    cin >> t;
    while (t--) {
        int n;
        cin >> n;
        string tasks;
        cin >> tasks;

        if (distracted(tasks)) {
            cout << "YES" << endl;
        } else {
            cout << "NO" << endl;
        }
    }
    return 0;
}
