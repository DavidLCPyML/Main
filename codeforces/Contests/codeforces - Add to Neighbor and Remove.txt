#include <iostream>
#include <vector>
#include <algorithm>
#include <tuple>
using namespace std;

void solve() {
    int n;
    cin >> n;
    vector<long long> a(n);
    long long sum = 0;
    for (long long& x : a) {
        cin >> x;
        sum += x;
    }

    for (int i = n; i >= 1; i--) {
        if (sum % i == 0) {
            long long needSum = sum / i;
            long long curSum = 0;
            bool ok = true;
            for (int j = 0; j < n; j++) {
                curSum += a[j];
                if (curSum > needSum) {
                    ok = false;
                    break;
                }
                else if (curSum == needSum) {
                    curSum = 0;
                }
            }

            if (ok) {
                cout << n - i << endl;
                return;
            }
        }
    }
}


int main() {
    int tests;
    cin >> tests;
    while (tests--) {
        solve();
    }
    return 0;
}
