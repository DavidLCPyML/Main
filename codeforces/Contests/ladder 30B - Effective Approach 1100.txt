        cin >> arr[i];
        pos[arr[i]] = i + 1;
    }

    long long int vasya = 0;
    long long int petya = 0;
    cin >> m;
    for (int i = 0; i < m; i++) {
        int query;
        cin >> query;
        vasya += pos[query];
        petya = petya + n - pos[query] + 1;
    }

    cout << vasya << " " << petya << endl;

    return 0;
