#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
	int t;
	cin >> t;
	while (t--) {
		int n;
		string s;
		cin >> n >> s;
		//aaabacadaeafagahaiajakalamanaoapaqarasatauavawaxayaz
		string smallest("a");
		while (s.find(smallest) != string::npos) {
			int size = smallest.size();
			
			//multiple digits
			while (size > 0 && smallest[size - 1] == 'z') {
				size--;
				//smallest[size - 1] = 'a';
			}
			//reset length
			if (size == 0) 
				smallest = string((int)smallest.size() + 1, 'a');
			//keep going
			else {
				smallest[size-1]++;
				for (int i = size; i < smallest.size(); i++) 
					smallest[i] = 'a';
			}
			//cout << smallest << endl;
		}
		cout << smallest << endl;
	}
	return 0;
}
