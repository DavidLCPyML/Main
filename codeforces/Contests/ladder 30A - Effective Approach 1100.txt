        long long int num;
        cin >> num;
        pos[num] = i+1;
    }

    long long int vasya = 0;
    long long int petya = 0;
    cin >> m;
    for (int i = 0; i < m; i++) {
        long long int query;
        cin >> query;
        vasya += pos[query];
        petya += n - pos[query]+1;
    }

    cout << vasya << " " << petya << "\n";

    return 0;
