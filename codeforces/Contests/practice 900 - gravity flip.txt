#include<iostream>
#include <vector>
#include <algorithm>
using namespace std;

void solve(int n) {
    vector<int> nums;
    for (int i = 0; i < n; i++) {
        int num;
        cin >> num;
        nums.push_back(num);
    }
    sort(nums.begin(), nums.end());

    for (int i = 0; i < n; i++) {
        cout << nums[i] << " ";
    }
    cout << endl;
}

int main() {
    int n;
    cin >> n;
    
    solve(n);
    return 0;
}
