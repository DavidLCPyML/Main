#include <iostream>
#include <cmath>
#include <algorithm>
#include <vector>
using namespace std;

int poly[300] = {0};
int mono[300] = {0};

void solve(int k, int n, int m) {
	vector<int> v;

	

	for (int i = 0; i < v.size(); i++) {
		if (v[i] > k + i) {
			cout << "-1" << endl;
			return;
		}
	}

	for (int i = 0; i < v.size(); i++) {
		cout << v[i] << " ";
	} cout << endl;

	return;
}

int main() {
	int t;
	cin >> t;

	while (t--) {
		int k, n, m;
		cin >> k >> n >> m;

		for (int i = 0; i < n; i++) {
			cin >> mono[i];
		}
		for (int i = 0; i < m; i++) {
			cin >> poly[i];
		}

		solve(k, n, m);
	}
	return 0;
}
