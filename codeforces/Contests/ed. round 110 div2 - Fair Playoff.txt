#include<iostream>
#include <vector>
#include <algorithm>
using namespace std;
vector<int> tourney(4,0);

void solve() {
    for (int i = 0; i < 4; i++) {
        cin >> tourney[i];
    }
    int adv1 = max(tourney[0], tourney[1]);
    int adv2 = max(tourney[2], tourney[3]);
    //cout << adv1 << " " << adv2 << endl;
    sort(tourney.begin(), tourney.end());
    //cout << tourney[2] << " " << tourney[3] << endl;
    
    if (adv1 == tourney[2] || adv1 == tourney[3]) {
        if (adv2 == tourney[2] || adv2 == tourney[3]) {
            cout << "YES" << endl;
            return;
        }
    }
    cout << "NO" << endl;
    return;
}

int main() {
    int t;
    cin >> t;
    while (t--) {
        solve();
    }
    return 0;
}
