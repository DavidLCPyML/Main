#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

void solve() {
    int n;
    int sum = 0;
    cin >> n;
    vector<int> arr;
    for (int i = 0; i < n; i++) {
        int x;
        cin >> x;
        arr.push_back(x);
        sum += x;
    }

    //cout << sum << " " << n << endl;
    //cout << endl;

    if (sum > n) {
        cout << sum - n << endl;
    }
    else if (sum == n) {
        cout << 0 << endl;
    }
    else {
        cout << 1 << endl;
    }
}



int main()
{
    int t;
    cin >> t;

    while (t--) {
        solve();
    }
    return 0;
}
