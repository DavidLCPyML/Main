#include <iostream>
#include <cmath>
using namespace std;

void solve(string s) {
	int arr[26] = { 0 };

	for (int i = 0; i < s.length(); i++) {
		arr[s[i] - 'a']++;

		if (arr[s[i] - 'a'] > 1) {
			cout << "NO" << endl;
			return;
		}
	}

	for (int i = 0; i < s.length(); i++) {
		if (arr[i] != 1) {
			cout << "NO" << endl;
			return;
		}
	}

	int a_pos = 0;
	for (int i = 0; i < s.length(); i++) {
		if (s[i] == 'a') {
			a_pos = i;
		}
	}

	for (int i = a_pos; i < s.length() - 1; i++) {
		if (s[i] >= s[i + 1]) {
			cout << "NO" << endl;
			return;
		}
	}

	for (int i = a_pos; i > 0; i--) {
		if (s[i - 1] <= s[i]) {
			cout << "NO" << endl;
			return;
		}
	}

	cout << "YES" << endl;
	return;
	/*for (int i = 0; i < 26; i++) {
		cout << arr[i] << " ";
	}cout << endl;*/
}

int main() {
	int n;
	cin >> n;

	while (n--) {
		string s;
		cin >> s;

		solve(s);
	}
	return 0;
}
