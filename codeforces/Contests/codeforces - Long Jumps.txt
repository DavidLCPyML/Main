#include <iostream>
#include <vector>
#include <algorithm>
#include <string>
using namespace std;

int main() {
    int tests;
    cin >> tests;
    while (tests--) {
        int n;
        cin >> n;
        vector<int> a(n);
        for (int i = 0; i < n; i++) {
            cin >> a[i];
        }

        vector<int> dp(n, 0);
        for (int i = n - 1; i >= 0; i--) {
            dp[i] = a[i];
            int j = i + a[i];
            if (j < n) {
                dp[i] += dp[j];
            }
        }
        cout << *max_element(dp.begin(), dp.end()) << endl;
    }
    return 0;
}
