#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
	int t;
	cin >> t;

	while (t--) {
		int n;
		cin >> n;
		vector<int> a(n);
		for (int i=0;i<n;i++)
			cin >> a[i];

		sort(a.begin(), a.end());
		if (a[0] < 0) {
			cout << "NO" << endl;
		} else {
			cout << "YES" << endl;
			cout << "300" << endl;
			for (int i = 0; i < 300; ++i) 
				cout << i << ' ';
			cout << endl;
		}
	}
	return 0;
}
