#include <iostream>
using namespace std;

int highestPowerof2(int n) {
    if (n < 1)
        return 0;
    int res = 1;
    // Try all powers starting from 2^1
    for (int i = 0; i < 8 * sizeof(unsigned int); i++) {
        int curr = 1 << i;
        if (curr > n)
            break;
        res = curr;
    }

    return res;
}

//biggest possible value of k is the largest power of 2 minus one
int main() {
    int t;
    cin >> t;
    while (t--) {
        int n;
        cin >> n;
        cout << highestPowerof2(n) - 1 << endl;
    }
    return 0;
}
