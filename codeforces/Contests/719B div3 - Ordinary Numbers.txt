#include <iostream>
using namespace std;

int countOrdinary(int n) {
    int mod = 1;
    int result = 0;
    int target = 9;

    for (int i = 10; i <= n; i *= 10) {
        target = i - 1;
        result += target / mod;
        mod = mod * 10 + 1;
    }
    result += n / mod;

    return result;
}

int main() {
    int t;
    cin >> t;

    while (t--) {
        int n;
        cin >> n;
        
        int result = countOrdinary(n);
        cout << result << endl;
    }
    return 0;
}
