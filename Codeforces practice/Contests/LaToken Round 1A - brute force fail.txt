#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
	int t;
	cin >> t;
	while (t--) {
		int n, m;
		cin >> n >> m;
		char arr[50][50];
		
		for (int i = 0; i < n; i++) {
			for (int j = 0; j < m; j++) {
				cin >> arr[i][j];
			}
		}
		
		/*for (int i = 0; i < n; i++) {
			for (int j = 0; j < m; j++) {
				cout << arr[i][j];
			}
			cout << endl;
		}*/

		char newarr[50][50];
		newarr[0][0] = 'R';
		for (int i = 0; i < n; i++) {
			if (newarr[i-1][0] == 'R')
				newarr[i][0] = 'W';
			else
				newarr[i][0] = 'R';

			for (int j = 1; j < m; j++) {
				if (newarr[i][j - 1] == 'R')
					newarr[i][j] = 'W';
				else
					newarr[i][j] = 'R';
			}
		}
		bool fits = true;
		for (int i = 0; i < n; i++) {
			for (int j = 0; j < m; j++) {
				if (newarr[i][j] != arr[i][j] && arr[i][j] != '.') {
					fits = false;
					break;
				}
			}
		}
		if (fits) {
			cout << "YES" << endl;
			for (int i = 0; i < n; i++) {
				for (int j = 0; j < m; j++) {
					cout << newarr[i][j];
				}
				cout << endl;
			}
		} else { 
			newarr[0][0] = 'W';
			for (int i = 0; i < n; i++) {
				if (newarr[i - 1][0] == 'W')
					newarr[i][0] = 'R';
				else
					newarr[i][0] = 'W';

				for (int j = 1; j < m; j++) {
					if (newarr[i][j - 1] == 'R')
						newarr[i][j] = 'W';
					else
						newarr[i][j] = 'R';
				}
			}
			
			fits = true;
			for (int i = 0; i < n; i++) {
				for (int j = 0; j < m; j++) {
					if (newarr[i][j] != arr[i][j] && arr[i][j] != '.') {
						fits = false;
						break;
					}
				}
			}
			if (fits) {
				cout << "YES" << endl;
				for (int i = 0; i < n; i++) {
					for (int j = 0; j < m; j++) {
						cout << newarr[i][j];
					}
					cout << endl;
				}
			}
			else {
				cout << "NO" << endl;
			}
		}
	}
	return 0;
}
