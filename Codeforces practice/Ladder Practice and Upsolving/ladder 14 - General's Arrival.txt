    for (int i = 0; i < n; i += 1) {
        cin >> line[i];

            max = line[i];
            maxpos = i;
        }
        if (min >= line[i]) {
            min = line[i];
            minpos = i;
        }
    }
    
    //cout << maxpos << " " << minpos << endl;

        cout << ans-1 << endl;
    } else {
        cout << ans << endl;
    }
