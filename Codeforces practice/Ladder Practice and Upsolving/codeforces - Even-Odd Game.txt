#include <iostream>
#include <vector>
#include <algorithm>
#include <string>
using namespace std;

int main() {
    int tests;
    cin >> tests;
    while (tests--) {
        int n;
        cin >> n;
        vector<int> a(n);
        for (int i = 0; i < n; i++) {
            cin >> a[i];
        }

        sort(a.rbegin(), a.rend());
        long long int asum = 0;
        long long int bsum = 0;

        /*for (int i = 0; i < n; i++) {
            cout << a[i] << " ";
        } cout << endl;
        */

        for (int i = 0; i < n; i++) {
            if (i % 2 == 0 && (a[i] % 2 == 0)) {
                asum += a[i];
            } else if (i % 2 == 1 && (a[i] % 2 == 1)){
                bsum += a[i];
            } else {
            }
        }

        if (asum > bsum) {
            cout << "Alice" << endl;
        } else if (asum < bsum) {
            cout << "Bob" << endl;
        } else {
            cout << "Tie" << endl;
        }
    }
    return 0;
}
