#include <iostream>
#include <vector>
using namespace std;

int N,M;

int main() {
	cin >> N >> M;
	vector<bool> sus(N+1, true);
	sus[0] = false;

	for (int i = 0; i < M; i++) {
		int tasker;
		cin >> tasker;
		//cout << sus[i] << " ";
		sus[tasker] = false;
	}

	/*
	for (int i = 0; i < N; i++) {
		cout << sus[i] << " ";
	}
	*/
	int sus_counter = 0;
	for (int i = 0; i < sus.size(); i++) {
		if (sus[i] == 1) {
			sus_counter++;
		}
	}

	if (sus_counter != 1) {
		cout << "NO" << endl;
	} else {
		cout << "YES" << endl;
	}
	return 0;
}
