#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

int main() {
	int n; 
	cin >> n;
	vector <int> v(n);
	int sum = 0;
	
	for (int I = 0; I < n; I++) {
		cin >> v[I]; 
		sum += v[I];
	}
	
	sort(v.begin(), v.end());
	reverse(v.begin(), v.end());
	int ans = 0, sm = 0;
	
	for (int I = 0; I < n; I++) {
		sm += v[I];
		ans++;
		if (sm > sum - sm)
			break;
	}

	cout << ans;
	return 0;
}
